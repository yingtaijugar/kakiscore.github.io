function exportToExcel() { commitActiveBoutToHistory(); if (globalHistoricalRegistry.length === 0) return alert("Historical data arrays are empty. Perform logging sequence before export processing."); const workbook = XLSX.utils.book_new(); // TAB ONE: GENERAL TOURNAMENT ROUNDS OVERVIEW SUMMARY const summaryDataMatrix = globalHistoricalRegistry.map(m => ({ "Match Number": m.matchLabel, "Home Team (A)": m.teamAKey, "Home Score": m.scoreA, "Away Score": m.scoreB, "Away Team (B)": m.teamBKey, "Total Live Actions Logged": m.timeline.length })); const summarySheet = XLSX.utils.json_to_sheet(summaryDataMatrix); XLSX.utils.book_append_sheet(workbook, summarySheet, "Tournament Overview"); // TAB TWO: DETAILED PERFORMANCE ACTIONS (DEDICATED STATS TRACKING) const performanceLogs = []; globalHistoricalRegistry.forEach(m => { if (m.timeline.length === 0) { performanceLogs.push({ "Match Instance": m.matchLabel, "Timestamp Reference": "N/A", "Affiliated Team": "N/A", "Event Action Type": "No Performance Tracked", "Player Name / Scorer": "N/A", "Assist Credit": "N/A" }); } else { // Parse events chronologically from game opening kickoff forward [...m.timeline].reverse().forEach(t => { performanceLogs.push({ "Match Instance": m.matchLabel, "Timestamp Reference": t.time, "Affiliated Team": t.team, "Event Action Type": t.type ? t.type.toUpperCase() : "GOAL", "Player Name / Scorer": t.player, "Assist Credit": t.type === "Goal" ? (t.assist || "Unassisted") : "N/A" }); }); } }); const detailingSheet = XLSX.utils.json_to_sheet(performanceLogs); XLSX.utils.book_append_sheet(workbook, detailingSheet, "Live Actions Log Matrix"); XLSX.writeFile(workbook, `Tournament_Performance_Registry.xlsx`); }
